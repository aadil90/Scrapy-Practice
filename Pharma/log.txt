    function send_email_com(form) {
        var error = false;

        var $button = $('#button_send_email_com');
        $button.prop('disabled', true);
        $button.removeClass('eventa');

        var s_url = "/ficha/json/send_form_com";

        var $wrapper = $(form);

        var message = $wrapper.find('p.message');

        var email = $wrapper.find('[name="email_user"]').val();
        var name_user = $wrapper.find('[name="name_user"]').val();
        var message_user = $wrapper.find('[ name="message_user"]').val();
        var id_com = $wrapper.parent().attr('data-id');
        var captcha = $wrapper.find('[name="captcha"]').val();

        $wrapper.find('input,textarea').each(function () {
            if ($(this).val() == '') {
                message.html(o_t.T_('Todos los campos son obligatorios')).addClass('error').removeClass('hide');
                error = true;
                return false;
            }
        });

        if (error === true) {
            $button.prop('disabled', false);
            return false;
        }

        if ($wrapper.find('[name="privacy_policy"]').prop('checked') == false) {
            message.html(o_t.T_('Para ponerte en contacto con esta empresa debes aceptar las condiciones del servicio.')).addClass('error').removeClass('hide');
            $button.prop('disabled', false);
            return false;
        }

        var o_data = {
            email_user: email,
            name_user: name_user,
            message_user: message_user,
            id_com: id_com,
            captcha: captcha
        };

        // Codificamos los datos
        var s_data = encode_b64_json(o_data);

        $.ajax({
            data: { base64: s_data },
            type: 'POST',
            dataType: 'json',
            url: s_url,
            success: function (o_res) {
                if (o_res.s_valid == 'OK') {
                    message.html(o_res.s_resp).removeClass('hide').addClass('success');
                    $button.hide();
                    $('#button_close_send_email').removeClass('pull-left').addClass('pull-right').text(o_t.T_('Volver'));
                    $wrapper.find('.hide_on_success').hide();
                }
                else {
                    message.html(o_res.s_resp).removeClass('hide').addClass('error');
                    $button.prop('disabled', false);
                }
            }
        });
    }

    function register_change_owner() {
        var $register_change_owner = $('#register_change_owner');
        var o_data = {
            register_button: $register_change_owner.find('.register_button'),
            message: $('.owner-info .message'),
            email_user: $register_change_owner.find('[name="email_user"]').val(),
            nick_user: $register_change_owner.find('[name="nick_user"]').val(),
            pass_user: $register_change_owner.find('[name="pass_user"]').val(),

            id_com: $register_change_owner.data('id_com'),

                return {
                    email_user: this.email_user,
                    nick_user: this.nick_user,
                    legal_chk: this.legal_chk,
                    pass_user: hex_sha1(this.pass_user),
                    id_com: this.id_com,

                    this.message.html(o_t.T_('El campo email es obligatorio'));
                    return this.b_error = true;
                }
                if (!isEmail(this.email_user)) {
                    this.message.html(o_t.T_('El email introducido no es válido'));
                    return this.b_error = true;
                }
                if (this.nick_user == '') {
                    this.message.html(o_t.T_('El nombre de usuario es obligatorio'));
                    return this.b_error = true;
                }

                if (this.pass_user == '') {
                    this.message.html(o_t.T_('La contraseña es obligatoria'));
                    return this.b_error = true;
                }

                if (this.pass_user != this.pass_user2) {
                    this.message.html(o_t.T_('Las contraseñas no coinciden. Introdúcelas nuevamente.'));
                    return this.b_error = true;

                if (this.owner_type.length == 0) {
                    this.message.html(o_t.T_('Selecciona tu relación con la empresa'));
                    return this.b_error = true;

                    return this.b_error = true;
                }
                else {
                    this.legal_chk = '1';
                }

        function publish_comment() {
            var $message = $('#message-comments-wrapper');
            var $publish = $('#publish_comment');
            var $textarea_comment = $('#textarea_comment');
            var s_comment = $.trim($textarea_comment.val());

            o_event_tracker.send_event($publish);

            if($publish.prop('disabled'))
                return false;

            if (s_comment == '') {
                alert('Write a comment');
                return false;
            }

            var o_data = {
                "id_com": 132025,
                "comment": s_comment
            };

            // Codificamos los datos
            var s_data = encode_b64_json(o_data);

            if (window.sessionStorage) {
                sessionStorage.setItem('s_comment', s_data);
            }

            $publish.prop('disabled', true);

            // Hacemos login/signup
            $.ajax({
                data: { base64: s_data },
                type: 'POST',
                dataType: 'json',
                url: "/ficha/json/send_comment",
                success: function (o_res) {
                    if (o_res.s_valid == 'OK') {
                        $publish.hide();
                        $('#textarea_comment').parent().hide();
                        $message.html('<span class="message success">' + o_res.s_resp + '</span>').removeClass('hide');
                        if(window.sessionStorage && sessionStorage.getItem('s_comment')) {
                            sessionStorage.removeItem('s_comment');
                        }
                    }
                    else {
                        if("Your session has expired. Please, log in again." == o_res.s_resp) {
                            $('#textarea_comment').parent().hide();
                            $('.rating').hide();
                            $publish.hide();

                            $message.html('<span class="message">Finally, to publish your opinion we need you to log in or sign up in Infoisinfo:</span>').removeClass('hide');
                            $('.comment-login').removeClass('hide');
                        }
                        else {
                            $publish.prop('disabled', false);
                            $message.html('<span class="message error">' + o_res.s_resp + '</span>').removeClass('hide');
                        }
                    }
                }
            });
        }

        function comment_limitter($textarea) {
            var n_limit = 1000;

            if(typeof $textarea.data('limit') !== 'undefined') {

            }

            var $counter = $('.rating .counter'),
                    $button = $('#publish_comment');

            var length = $textarea.val().length;
            $counter.children('.actual').text(length);


            if (length > n_limit) {
                $counter.addClass('exceeded');
                $textarea.addClass('exceeded');
                $button.prop('disabled', true);
            } else {
                $counter.removeClass('exceeded');
                $textarea.removeClass('exceeded');
                $button.prop('disabled', false);
            }
